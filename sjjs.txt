package main

import (
	"encoding/json"
	"fmt"
	"math"
	"os"
	"os/exec"
	"sort"

	"github.com/jezek/xgb"
	"github.com/jezek/xgb/xproto"
)

type Config struct {
	TerminalKey    byte `json:"terminal_key"`
	RingLeftKey    byte `json:"ring_left_key"`
	RingRightKey   byte `json:"ring_right_key"`
	RingModeKey    byte `json:"ring_mode_key"`
	CloseWindowKey byte `json:"close_window_key"`
	DmenuKey       byte `json:"dmenu_key"`
	ExitWmKey      byte `json:"exit_wm_key"`
}

type WindowManager struct {
	X                *xgb.Conn
	Root             xproto.Window
	ScreenWidth      uint16
	ScreenHeight     uint16
	Workspaces       [][]xproto.Window
	CurrentWorkspace int
	RingMode         bool
	CurrentIndex     int
	ZoomMode         bool
	Cfg              Config
	Gaps             int
	WmProtocols      xproto.Atom
	WmDeleteWindow   xproto.Atom
}

func loadConfig() Config {
	defaultCfg := Config{
		TerminalKey:    36,
		RingLeftKey:    113,
		RingRightKey:   114,
		RingModeKey:    27,
		CloseWindowKey: 38,
		DmenuKey:       40,
		ExitWmKey:      58,
	}
	file, err := os.Open("config.json")
	if err != nil {
		return defaultCfg
	}
	defer file.Close()
	var cfg Config
	_ = json.NewDecoder(file).Decode(&cfg)
	return cfg
}

func main() {
	X, err := xgb.NewConn()
	if err != nil {
		fmt.Printf("Ошибка подключения к X: %v\n", err)
		return
	}
	defer X.Close()

	setup := xproto.Setup(X)
	screen := setup.DefaultScreen(X)

	protocolsReply, _ := xproto.InternAtom(X, false, uint16(len("WM_PROTOCOLS")), "WM_PROTOCOLS").Reply()
	deleteReply, _ := xproto.InternAtom(X, false, uint16(len("WM_DELETE_WINDOW")), "WM_DELETE_WINDOW").Reply()

	wm := &WindowManager{
		X:                X,
		Root:             screen.Root,
		ScreenWidth:      screen.WidthInPixels,
		ScreenHeight:     screen.HeightInPixels,
		Workspaces:       make([][]xproto.Window, 9),
		CurrentWorkspace: 0,
		RingMode:         false,
		CurrentIndex:     0,
		ZoomMode:         false,
		Cfg:              loadConfig(),
		Gaps:             5,
		WmProtocols:      protocolsReply.Atom,
		WmDeleteWindow:   deleteReply.Atom,
	}

	for i := 0; i < 9; i++ {
		wm.Workspaces[i] = make([]xproto.Window, 0)
	}

	err = xproto.ChangeWindowAttributesChecked(X, wm.Root, xproto.CwEventMask,
		[]uint32{xproto.EventMaskSubstructureRedirect | xproto.EventMaskSubstructureNotify}).Check()
	if err != nil {
		fmt.Println("Другой оконный менеджер уже запущен или нет прав.")
		return
	}

	wm.grabKey(wm.Cfg.TerminalKey, xproto.ModMask4)
	wm.grabKey(wm.Cfg.RingLeftKey, xproto.ModMask4)
	wm.grabKey(wm.Cfg.RingRightKey, xproto.ModMask4)
	wm.grabKey(wm.Cfg.RingModeKey, xproto.ModMask4)
	wm.grabKey(wm.Cfg.CloseWindowKey, xproto.ModMask4)
	wm.grabKey(wm.Cfg.DmenuKey, xproto.ModMask4)
	wm.grabKey(wm.Cfg.ExitWmKey, xproto.ModMask4|xproto.ModMaskShift)
	wm.grabKey(111, xproto.ModMask4) // Up
	wm.grabKey(116, xproto.ModMask4) // Down

	for i := byte(10); i <= 18; i++ {
		wm.grabKey(i, xproto.ModMask4)
	}

	fmt.Println("RingWM запущен")

	for {
		ev, err := X.WaitForEvent()
		if err != nil {
			continue
		}
		if ev == nil {
			break
		}

		switch e := ev.(type) {
		case xproto.MapRequestEvent:
			wm.manageWindow(e.Window)
		case xproto.ConfigureRequestEvent:
			wm.handleConfigureRequest(e)
		case xproto.UnmapNotifyEvent:
			wm.unmanageWindow(e.Window)
		case xproto.DestroyNotifyEvent:
			wm.unmanageWindow(e.Window)
		case xproto.KeyPressEvent:
			if shouldQuit := wm.handleKeyPress(e.Detail, e.State); shouldQuit {
				fmt.Println("Выход...")
				return
			}
		}
	}
}

func (wm *WindowManager) handleConfigureRequest(e xproto.ConfigureRequestEvent) {
	mask := e.ValueMask
	vals := make([]uint32, 0)
	if mask&xproto.ConfigWindowX != 0 {
		vals = append(vals, uint32(e.X))
	}
	if mask&xproto.ConfigWindowY != 0 {
		vals = append(vals, uint32(e.Y))
	}
	if mask&xproto.ConfigWindowWidth != 0 {
		vals = append(vals, uint32(e.Width))
	}
	if mask&xproto.ConfigWindowHeight != 0 {
		vals = append(vals, uint32(e.Height))
	}
	if mask&xproto.ConfigWindowBorderWidth != 0 {
		vals = append(vals, uint32(e.BorderWidth))
	}
	if mask&xproto.ConfigWindowSibling != 0 {
		vals = append(vals, uint32(e.Sibling))
	}
	if mask&xproto.ConfigWindowStackMode != 0 {
		vals = append(vals, uint32(e.StackMode))
	}
	_ = xproto.ConfigureWindowChecked(wm.X, e.Window, mask, vals).Check()
}

func (wm *WindowManager) grabKey(keycode byte, modifiers uint16) {
	xproto.GrabKey(wm.X, true, wm.Root, modifiers, xproto.Keycode(keycode), xproto.GrabModeAsync, xproto.GrabModeAsync)
}

func (wm *WindowManager) getActiveWindows() []xproto.Window {
	return wm.Workspaces[wm.CurrentWorkspace]
}

func (wm *WindowManager) setActiveWindows(wins []xproto.Window) {
	wm.Workspaces[wm.CurrentWorkspace] = wins
}

func (wm *WindowManager) handleKeyPress(keycode xproto.Keycode, state uint16) bool {
	code := byte(keycode)

	if code == wm.Cfg.ExitWmKey && (state&xproto.ModMask4) != 0 && (state&xproto.ModMaskShift) != 0 {
		return true
	}
	if code >= 10 && code <= 18 && (state&xproto.ModMaskShift) == 0 {
		targetWorkspace := int(code - 10)
		if targetWorkspace != wm.CurrentWorkspace {
			wm.CurrentWorkspace = targetWorkspace
			wm.RingMode = false
			wm.ZoomMode = false
			wm.arrangeWindows()
		}
		return false
	}

	switch code {
	case wm.Cfg.TerminalKey:
		_ = exec.Command("alacritty").Start()
	case wm.Cfg.DmenuKey:
		_ = exec.Command("dmenu_run").Start()
	case wm.Cfg.CloseWindowKey:
		wm.killActiveWindow()
	case wm.Cfg.RingModeKey:
		if len(wm.getActiveWindows()) > 0 {
			wm.RingMode = !wm.RingMode
			wm.ZoomMode = false
			wm.arrangeWindows()
		}
	case wm.Cfg.RingRightKey:
		wins := wm.getActiveWindows()
		if wm.RingMode && len(wins) > 0 {
			wm.CurrentIndex = (wm.CurrentIndex + 1) % len(wins)
			wm.ZoomMode = false
			wm.arrangeWindows()
		}
	case wm.Cfg.RingLeftKey:
		wins := wm.getActiveWindows()
		if wm.RingMode && len(wins) > 0 {
			wm.CurrentIndex = (wm.CurrentIndex - 1 + len(wins)) % len(wins)
			wm.ZoomMode = false
			wm.arrangeWindows()
		}
	case 111:
		if wm.RingMode {
			wm.ZoomMode = true
			wm.arrangeWindows()
		}
	case 116:
		if wm.RingMode {
			wm.ZoomMode = false
			wm.arrangeWindows()
		}
	}
	return false
}

func (wm *WindowManager) killActiveWindow() {
	wins := wm.getActiveWindows()
	if len(wins) == 0 {
		return
	}
	target := wins[len(wins)-1]
	if wm.RingMode {
		target = wins[wm.CurrentIndex]
	}
	event := xproto.ClientMessageEvent{
		Format: 32,
		Window: target,
		Type:   wm.WmProtocols,
	}
	deleteAtom := uint32(wm.WmDeleteWindow)
	event.Data.Bytes()[0] = byte(deleteAtom)
	event.Data.Bytes()[1] = byte(deleteAtom >> 8)
	event.Data.Bytes()[2] = byte(deleteAtom >> 16)
	event.Data.Bytes()[3] = byte(deleteAtom >> 24)
	err := xproto.SendEventChecked(wm.X, false, target, xproto.EventMaskNoEvent, string(event.Bytes())).Check()
	if err != nil {
		xproto.DestroyWindow(wm.X, target)
	}
}

func (wm *WindowManager) manageWindow(win xproto.Window) {
	attrs, _ := xproto.GetWindowAttributes(wm.X, win).Reply()
	if attrs != nil && attrs.OverrideRedirect {
		xproto.MapWindow(wm.X, win)
		return
	}
	for _, w := range wm.getActiveWindows() {
		if w == win {
			return
		}
	}
	xproto.ChangeWindowAttributes(wm.X, win, xproto.CwEventMask, []uint32{xproto.EventMaskEnterWindow | xproto.EventMaskFocusChange})
	wins := wm.getActiveWindows()
	wins = append(wins, win)
	wm.setActiveWindows(wins)
	wm.CurrentIndex = len(wins) - 1
	xproto.MapWindow(wm.X, win)
	wm.arrangeWindows()
}

func (wm *WindowManager) unmanageWindow(win xproto.Window) {
	for idx := 0; idx < 9; idx++ {
		wins := wm.Workspaces[idx]
		for i, w := range wins {
			if w == win {
				wins = append(wins[:i], wins[i+1:]...)
				wm.Workspaces[idx] = wins
				if idx == wm.CurrentWorkspace {
					if len(wins) == 0 {
						wm.RingMode = false
						wm.ZoomMode = false
						wm.CurrentIndex = 0
					} else {
						wm.CurrentIndex = wm.CurrentIndex % len(wins)
					}
					wm.arrangeWindows()
				}
				return
			}
		}
	}
}

type RenderWindow struct {
	ID      xproto.Window
	X, Y    int
	W, H    int
	ZDepth  float64
	IsFocus bool
}

func (wm *WindowManager) arrangeWindows() {
	wins := wm.getActiveWindows()
	n := len(wins)
	if n == 0 {
		return
	}
	cx := int(wm.ScreenWidth) / 2
	cy := int(wm.ScreenHeight) / 2

	if wm.RingMode {
		if wm.ZoomMode && n > 0 {
			targetWin := wins[wm.CurrentIndex]
			zw := int(wm.ScreenWidth) - 2*wm.Gaps
			zh := int(wm.ScreenHeight) - 2*wm.Gaps
			zx := wm.Gaps
			zy := wm.Gaps
			_ = xproto.ConfigureWindowChecked(wm.X, targetWin,
				xproto.ConfigWindowX|xproto.ConfigWindowY|xproto.ConfigWindowWidth|xproto.ConfigWindowHeight,
				[]uint32{uint32(zx), uint32(zy), uint32(zw), uint32(zh)}).Check()
			for i, win := range wins {
				if i != wm.CurrentIndex {
					_ = xproto.UnmapWindow(wm.X, win)
				} else {
					_ = xproto.MapWindow(wm.X, win)
				}
			}
			xproto.SetInputFocus(wm.X, xproto.InputFocusPointerRoot, targetWin, xproto.TimeCurrentTime)
			return
		}
		for _, win := range wins {
			xproto.MapWindow(wm.X, win)
		}
		radiusX := int(float64(wm.ScreenWidth) * 0.35)
		radiusY := int(float64(wm.ScreenHeight) * 0.22)
		queue := make([]RenderWindow, n)
		for i, win := range wins {
			angle := (2.0*math.Pi*float64(i-wm.CurrentIndex))/float64(n) + math.Pi/2.0
			zDepth := math.Sin(angle)
			scale := 0.4 + (zDepth+1.0)*0.2
			ww := int(float64(wm.ScreenWidth) * 0.45 * scale)
			wh := int(float64(wm.ScreenHeight) * 0.45 * scale)
			x := int(float64(cx)+float64(radiusX)*math.Cos(angle)) - ww/2
			y := int(float64(cy)+float64(radiusY)*math.Sin(angle)) - wh/2
			queue[i] = RenderWindow{ID: win, X: x, Y: y, W: ww, H: wh, ZDepth: zDepth, IsFocus: i == wm.CurrentIndex}
		}
		sort.Slice(queue, func(i, j int) bool { return queue[i].ZDepth > queue[j].ZDepth })
		for _, rw := range queue {
			_ = xproto.ConfigureWindowChecked(wm.X, rw.ID,
				xproto.ConfigWindowX|xproto.ConfigWindowY|xproto.ConfigWindowWidth|xproto.ConfigWindowHeight,
				[]uint32{uint32(rw.X), uint32(rw.Y), uint32(rw.W), uint32(rw.H)}).Check()
			if rw.IsFocus {
				xproto.SetInputFocus(wm.X, xproto.InputFocusPointerRoot, rw.ID, xproto.TimeCurrentTime)
			}
		}
	} else {
		sw := int(wm.ScreenWidth)
		sh := int(wm.ScreenHeight)
		cnt := len(wins)
		for i, win := range wins {
			xproto.MapWindow(wm.X, win)
			switch cnt {
			case 1:
				_ = xproto.ConfigureWindowChecked(wm.X, win,
					xproto.ConfigWindowX|xproto.ConfigWindowY|xproto.ConfigWindowWidth|xproto.ConfigWindowHeight,
					[]uint32{0, 0, uint32(sw), uint32(sh)}).Check()
			case 2:
				if i == 0 {
					_ = xproto.ConfigureWindowChecked(wm.X, win,
						xproto.ConfigWindowX|xproto.ConfigWindowY|xproto.ConfigWindowWidth|xproto.ConfigWindowHeight,
						[]uint32{0, 0, uint32(sw / 2), uint32(sh)}).Check()
				} else {
					_ = xproto.ConfigureWindowChecked(wm.X, win,
						xproto.ConfigWindowX|xproto.ConfigWindowY|xproto.ConfigWindowWidth|xproto.ConfigWindowHeight,
						[]uint32{uint32(sw / 2), 0, uint32(sw / 2), uint32(sh)}).Check()
				}
			case 3:
				if i == 0 {
					_ = xproto.ConfigureWindowChecked(wm.X, win,
						xproto.ConfigWindowX|xproto.ConfigWindowY|xproto.ConfigWindowWidth|xproto.ConfigWindowHeight,
						[]uint32{0, 0, uint32(sw / 2), uint32(sh)}).Check()
				} else if i == 1 {
					_ = xproto.ConfigureWindowChecked(wm.X, win,
						xproto.ConfigWindowX|xproto.ConfigWindowY|xproto.ConfigWindowWidth|xproto.ConfigWindowHeight,
						[]uint32{uint32(sw / 2), 0, uint32(sw / 2), uint32(sh / 2)}).Check()
				} else {
					_ = xproto.ConfigureWindowChecked(wm.X, win,
						xproto.ConfigWindowX|xproto.ConfigWindowY|xproto.ConfigWindowWidth|xproto.ConfigWindowHeight,
						[]uint32{uint32(sw / 2), uint32(sh / 2), uint32(sw / 2), uint32(sh / 2)}).Check()
				}
			default:
				cols := 2
				rows := (cnt + 1) / 2
				colW := sw / cols
				rowH := sh / rows
				row := i / cols
				col := i % cols
				_ = xproto.ConfigureWindowChecked(wm.X, win,
					xproto.ConfigWindowX|xproto.ConfigWindowY|xproto.ConfigWindowWidth|xproto.ConfigWindowHeight,
					[]uint32{uint32(col*colW), uint32(row * rowH), uint32(colW), uint32(rowH)}).Check()
			}
		}
	}
}
